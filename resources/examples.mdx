## Platform Initialization

<CodeGroup>

```kotlin Android
import com.dadomatch.shared.initKoinAndroid

class MyApplication : Application() {
    override fun onCreate() {
        super.onCreate()
        
        // Initialize the shared SDK
        initKoinAndroid(this)
    }
}
```

```swift iOS
import DadoMatchShared

class AppDelegate: UIResponder, UIApplicationDelegate {
    func application(
        _ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
    ) -> Bool {
        // Initialize Koin dependency injection
        KoinInitializerKt.initKoin()
        return true
    }
}
```

</CodeGroup>

<Note>
  On Android, the SDK automatically initializes Firebase and database components. On iOS, ensure you call `initKoin()` before accessing any shared functionality.
</Note>

## Authentication

### Implementing Native Auth Handlers

<CodeGroup>

```kotlin Android
import com.dadomatch.shared.feature.auth.presentation.NativeAuthHandler
import com.dadomatch.shared.feature.auth.presentation.GoogleTokens
import com.dadomatch.shared.feature.auth.presentation.AuthTokens

class AndroidAuthHandler(
    private val activity: FragmentActivity
) : NativeAuthHandler {
    
    override suspend fun signInWithGoogle(): Result<GoogleTokens> {
        return try {
            // Implement Google Sign-In using Credential Manager
            val credential = credentialManager.getCredential(
                request = getGoogleIdOption(context),
                context = context
            )
            
            val googleIdTokenCredential = GoogleIdTokenCredential
                .createFrom(credential.credential.data)
            
            Result.success(
                GoogleTokens(
                    idToken = googleIdTokenCredential.idToken,
                    accessToken = null // Android doesn't require access token
                )
            )
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    
    override suspend fun signInWithApple(): Result<AuthTokens> {
        // Apple Sign-In implementation
        return Result.failure(Exception("Not implemented on Android"))
    }
}
```

```swift iOS
import DadoMatchShared
import AuthenticationServices
import GoogleSignIn

class IosAuthHandler: NativeAuthHandler {
    
    func signInWithGoogle() async -> ResultOf<GoogleTokens> {
        do {
            let result = try await GIDSignIn.sharedInstance.signIn(
                withPresenting: getRootViewController()
            )
            
            guard let idToken = result.user.idToken?.tokenString,
                  let accessToken = result.user.accessToken.tokenString else {
                return ResultOf.failure(
                    error: KotlinThrowable(message: "Missing tokens")
                )
            }
            
            return ResultOf.success(
                value: GoogleTokens(
                    idToken: idToken,
                    accessToken: accessToken
                )
            )
        } catch {
            return ResultOf.failure(error: error.asKotlinThrowable())
        }
    }
    
    func signInWithApple() async -> ResultOf<AuthTokens> {
        // Apple Sign-In implementation
        // ...
    }
}
```

</CodeGroup>

<Warning>
  Platform-specific auth handlers must be registered before using authentication features. The shared SDK provides the interface, but native implementations are required for each platform.
</Warning>

## Subscription Management

### Checking Premium Status

```kotlin
import com.dadomatch.shared.feature.subscription.data.remote.RevenueCatService
import org.koin.core.component.KoinComponent
import org.koin.core.component.inject

class SubscriptionManager : KoinComponent {
    private val revenueCatService: RevenueCatService by inject()
    
    suspend fun checkPremiumAccess(): Boolean {
        return when (val result = revenueCatService.getCustomerInfo()) {
            is Result.Success -> {
                revenueCatService.hasPremiumAccess(result.data)
            }
            is Result.Failure -> false
        }
    }
}
```

### Purchasing a Subscription

```kotlin
import com.dadomatch.shared.feature.subscription.domain.usecase.PurchaseSubscriptionUseCase
import com.dadomatch.shared.feature.subscription.domain.usecase.GetAvailableProductsUseCase

class MyViewModel : ViewModel() {
    private val getProductsUseCase: GetAvailableProductsUseCase = get()
    private val purchaseUseCase: PurchaseSubscriptionUseCase = get()
    
    fun purchasePremium() {
        viewModelScope.launch {
            // Get available products
            val products = getProductsUseCase()
            val premiumProduct = products.firstOrNull { it.id == "premium_monthly" }
            
            premiumProduct?.let { product ->
                // Purchase the subscription
                when (val result = purchaseUseCase(product)) {
                    is Resource.Success -> {
                        // Handle success
                    }
                    is Resource.Error -> {
                        // Handle error
                    }
                }
            }
        }
    }
}
```

## Database Operations

### Working with Local Data

```kotlin
import com.dadomatch.shared.feature.success.domain.usecase.AddSuccessUseCase
import com.dadomatch.shared.feature.success.domain.usecase.GetSuccessesUseCase
import com.dadomatch.shared.feature.success.domain.model.SuccessRecord

class SuccessTracker : KoinComponent {
    private val addSuccessUseCase: AddSuccessUseCase by inject()
    private val getSuccessesUseCase: GetSuccessesUseCase by inject()
    
    suspend fun trackSuccess(title: String, description: String) {
        val success = SuccessRecord(
            id = 0,
            title = title,
            description = description,
            timestamp = Clock.System.now().toEpochMilliseconds()
        )
        
        addSuccessUseCase(success)
    }
    
    suspend fun getAllSuccesses(): List<SuccessRecord> {
        return when (val result = getSuccessesUseCase()) {
            is Resource.Success -> result.data
            is Resource.Error -> emptyList()
        }
    }
}
```
