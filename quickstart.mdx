# Quickstart

This guide will get you up and running with DadoMatch Shared in your Android or iOS app. You'll initialize Koin, set up authentication, and make your first API call.

## Prerequisites

Before you begin, make sure you have:

- [Installed DadoMatch Shared](/installation) in your project
- Firebase project configured (for authentication)
- RevenueCat account configured (for subscriptions)

## Platform Setup

<Tabs>
  <Tab title="Android">

### Android Quickstart

<Steps>
  <Step title="Initialize Koin in your Application class">
    Create or update your `Application` class to initialize Koin:

    ```kotlin MyApplication.kt
    package com.example.myapp

    import android.app.Application
    import com.dadomatch.shared.initKoin

    class MyApplication : Application() {
        override fun onCreate() {
            super.onCreate()
            
            // Initialize Koin with Android context
            initKoin {
                androidContext(this@MyApplication)
            }
        }
    }
    ```

    Register your Application class in `AndroidManifest.xml`:

    ```xml AndroidManifest.xml
    <application
        android:name=".MyApplication"
        android:label="@string/app_name"
        android:theme="@style/AppTheme">
        <!-- ... -->
    </application>
    ```
  </Step>

  <Step title="Configure environment variables">
    Create a `local.properties` file in your project root:

    ```properties local.properties
    GEMINI_API_KEY=your_gemini_api_key_here
    GEMINI_MODEL_NAME=gemini-2.5-flash-lite
    GOOGLE_WEB_CLIENT_ID=your_google_client_id
    STAGE_REVENUECAT_API_KEY=your_revenuecat_stage_key
    PROD_REVENUECAT_API_KEY=your_revenuecat_prod_key
    STAGE_API_BASE_URL=https://api-stage.dadomatch.com
    PROD_API_BASE_URL=https://api.dadomatch.com
    app.flavor=stage
    ```

    <Warning>
      Never commit `local.properties` to version control. Add it to `.gitignore`.
    </Warning>
  </Step>

  <Step title="Use authentication in your UI">
    Get the `AuthViewModel` from Koin and observe authentication state:

    ```kotlin MainActivity.kt
    package com.example.myapp

    import androidx.activity.ComponentActivity
    import androidx.activity.compose.setContent
    import androidx.compose.material3.Button
    import androidx.compose.material3.Text
    import androidx.compose.runtime.collectAsState
    import androidx.compose.runtime.getValue
    import com.dadomatch.shared.feature.auth.presentation.viewmodel.AuthViewModel
    import org.koin.androidx.compose.koinViewModel

    class MainActivity : ComponentActivity() {
        override fun onCreate(savedInstanceState: Bundle?) {
            super.onCreate(savedInstanceState)
            setContent {
                val authViewModel: AuthViewModel = koinViewModel()
                val uiState by authViewModel.uiState.collectAsState()

                when {
                    uiState.user != null -> {
                        Text("Welcome, ${uiState.user?.displayName ?: "User"}!")
                    }
                    else -> {
                        Button(onClick = { authViewModel.triggerGoogleSignIn() }) {
                            Text("Sign in with Google")
                        }
                    }
                }
            }
        }
    }
    ```
  </Step>

  <Step title="Generate an icebreaker">
    Use the icebreaker generation feature:

    ```kotlin IcebreakerExample.kt
    import com.dadomatch.shared.feature.icebreaker.domain.usecase.GenerateIcebreakerUseCase
    import com.dadomatch.shared.core.util.Resource
    import org.koin.core.component.KoinComponent
    import org.koin.core.component.inject

    class IcebreakerExample : KoinComponent {
        private val generateIcebreaker: GenerateIcebreakerUseCase by inject()

        suspend fun getIcebreaker() {
            val result = generateIcebreaker(
                environment = "Casual hangout",
                intensity = "Light",
                language = "en"
            )

            when (result) {
                is Resource.Success -> {
                    println("Icebreaker: ${result.data}")
                }
                is Resource.Error -> {
                    println("Error: ${result.message}")
                }
                is Resource.Loading -> {
                    println("Loading...")
                }
            }
        }
    }
    ```
  </Step>
</Steps>

### iOS Quickstart

<Steps>
  <Step title="Initialize Koin in your App">
    Initialize Koin when your app launches:

    ```swift MyApp.swift
    import SwiftUI
    import DadoMatchShared

    @main
    struct MyApp: App {
        init() {
            // Initialize Koin
            KoinInitializerKt.initKoin()
        }
        
        var body: some Scene {
            WindowGroup {
                ContentView()
            }
        }
    }
    ```
  </Step>

  <Step title="Configure environment variables">
    The iOS package uses the same BuildKonfig configuration as Android. Environment variables are baked into the XCFramework at build time.

    <Note>
      For local development, configure the variables in the shared module's `local.properties` before building the XCFramework.
    </Note>
  </Step>

  <Step title="Use authentication in your UI">
    Access the `AuthViewModel` through Koin:

    ```swift ContentView.swift
    import SwiftUI
    import DadoMatchShared

    struct ContentView: View {
        @StateObject private var authObserver = AuthViewModelObserver()
        
        var body: some View {
            if let user = authObserver.user {
                Text("Welcome, \\(user.displayName ?? "User")!")
            } else {
                Button("Sign in with Google") {
                    authObserver.viewModel.triggerGoogleSignIn()
                }
            }
        }
    }

    // Helper to observe Kotlin Flow in SwiftUI
    class AuthViewModelObserver: ObservableObject {
        @Published var user: AuthUser?
        let viewModel: AuthViewModel
        
        init() {
            let koin = KoinPlatformToolsKt.defaultContext().get()
            viewModel = koin.get(objCClass: AuthViewModel.self) as! AuthViewModel
            
            // Observe the uiState Flow
            viewModel.uiState.collect(collector: Collector { state in
                DispatchQueue.main.async {
                    self.user = (state as? AuthUiState)?.user
                }
            }) { error in
                print("Error collecting state: \\(error)")
            }
        }
    }
    ```
  </Step>

  <Step title="Generate an icebreaker">
    Call the icebreaker generation use case:

    ```swift IcebreakerView.swift
    import DadoMatchShared

    class IcebreakerViewModel: ObservableObject {
        @Published var icebreaker: String = ""
        private let generateUseCase: GenerateIcebreakerUseCase
        
        init() {
            let koin = KoinPlatformToolsKt.defaultContext().get()
            generateUseCase = koin.get(objCClass: GenerateIcebreakerUseCase.self) 
                as! GenerateIcebreakerUseCase
        }
        
        func generate() async {
            let result = await generateUseCase.invoke(
                environment: "Casual hangout",
                intensity: "Light",
                language: "en"
            )
            
            if let success = result as? ResourceSuccess<NSString> {
                DispatchQueue.main.async {
                    self.icebreaker = success.data as String
                }
            }
        }
    }
    ```
  </Step>
</Steps>
