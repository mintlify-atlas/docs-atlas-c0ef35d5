The DadoMatch Shared SDK provides a complete subscription management system powered by RevenueCat, with built-in paywall UI, entitlement checking, and purchase restoration.

## Architecture Overview

The subscription system follows clean architecture:

- **SubscriptionRepository**: Domain interface for subscription operations
- **SubscriptionViewModel**: Manages subscription state across the app
- **PaywallScreen**: Pre-built RevenueCat paywall integration
- **CheckEntitlementUseCase**: Validates premium feature access

## Quick Start

<Steps>

## Subscription Models

### SubscriptionTier

enum class SubscriptionTier {

## Repository Interface

The `SubscriptionRepository` provides all subscription operations:

```kotlin
interface SubscriptionRepository {
    // Observe subscription status
    fun getSubscriptionStatus(): Flow<SubscriptionStatus>
    
    // Get current status (one-time)
    suspend fun getCurrentSubscriptionStatus(): Result<SubscriptionStatus>
    
    // Purchase a subscription
    suspend fun purchaseSubscription(productId: String): Result<SubscriptionStatus>
    
    // Restore previous purchases
    suspend fun restorePurchases(): Result<SubscriptionStatus>
    
    // Check entitlement
    suspend fun hasEntitlement(entitlement: Entitlement): Boolean
    
    // Get available products
    suspend fun getAvailableProducts(): Result<List<Product>>
    
    // Roll management
    suspend fun decrementDailyRolls(): Result<SubscriptionStatus>
    suspend fun resetDailyRolls()
    
    // User linking
    suspend fun logIn(userId: String): Result<SubscriptionStatus>
    suspend fun logOut(): Result<SubscriptionStatus>
}
```

## Use Cases

### GetSubscriptionStatusUseCase

Observe or fetch subscription status:

```kotlin
class GetSubscriptionStatusUseCase(
    private val subscriptionRepository: SubscriptionRepository
) {
    // Reactive flow
    operator fun invoke(): Flow<SubscriptionStatus> {
        return subscriptionRepository.getSubscriptionStatus()
    }
    
    // One-time fetch
    suspend fun getCurrentStatus(): Result<SubscriptionStatus> {
        return subscriptionRepository.getCurrentSubscriptionStatus()
    }
}
```

**Usage Example:**

```kotlin
class MyViewModel(
    private val getSubscriptionStatus: GetSubscriptionStatusUseCase
) : ViewModel() {
    
    init {
        viewModelScope.launch {
            // Observe changes
            getSubscriptionStatus().collect { status ->
                println("Subscription tier: ${status.tier}")
                println("Rolls remaining: ${status.dailyRollsRemaining}")
            }
        }
    }
    
    suspend fun checkCurrentStatus() {
        getSubscriptionStatus.getCurrentStatus()
            .onSuccess { status ->
                println("Current tier: ${status.tier}")
            }
            .onFailure { error ->
                println("Error: ${error.message}")
            }
    }
}
```

### CheckEntitlementUseCase

Validate access to premium features:

```kotlin
class CheckEntitlementUseCase(
    val subscriptionRepository: SubscriptionRepository
) {
    suspend fun hasEntitlement(entitlement: Entitlement): Boolean {
        return subscriptionRepository.hasEntitlement(entitlement)
    }
    
    fun canAccessCategory(intensity: String): Boolean {
        // Spicy category requires premium
        return if (intensity == "int_spicy") {
            subscriptionRepository.hasEntitlement(Entitlement.SPICY_CATEGORY)
        } else {
            true  // Other categories are free
        }
    }
}
```

## PaywallScreen Component

The SDK includes a fully functional paywall screen with RevenueCat integration:

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun PaywallScreen(onDismiss: () -> Unit = {}) {
    val viewModel: SubscriptionViewModel = koinViewModel()
    val authViewModel: AuthViewModel = koinViewModel()
    
    val status by viewModel.subscriptionStatus.collectAsState()
    val authUiState by authViewModel.uiState.collectAsState()
    
    val isAnonymous = authUiState.user?.isAnonymous ?: true
    
    Box(modifier = Modifier.fillMaxSize().background(DeepDarkBlue)) {
        if (isAnonymous) {
            // Show login prompt for anonymous users
            LoginPrompt(
                onLogin = { /* Show auth sheet */ },
                onDismiss = onDismiss
            )
        } else {
            // Show RevenueCat paywall
            Paywall(
                options = PaywallOptions(
                    dismissRequest = onDismiss
                ) {
                    shouldDisplayDismissButton = true
                    listener = object : PaywallListener {
                        override fun onPurchaseCompleted(
                            customerInfo: CustomerInfo,
                            storeTransaction: StoreTransaction
                        ) {
                            viewModel.refreshStatus()
                        }
                        
                        override fun onRestoreCompleted(customerInfo: CustomerInfo) {
                            viewModel.refreshStatus()
                        }
                    }
                }
            )
        }
    }
}
```

<Note>
  The `PaywallScreen` automatically handles authentication requirements. Anonymous users must sign in before purchasing.
</Note>

## Daily Roll Management

Free users have limited daily rolls:

```kotlin
class RollDiceUseCase(
    private val subscriptionRepository: SubscriptionRepository
) {
    suspend operator fun invoke(): Result<Unit> {
        val status = subscriptionRepository.getCurrentSubscriptionStatus()
            .getOrNull() ?: return Result.failure(Exception("No status"))
        
        // Premium users have unlimited rolls
        if (status.tier == SubscriptionTier.PREMIUM) {
            return Result.success(Unit)
        }
        
        // Check if free user has rolls remaining
        val remaining = status.dailyRollsRemaining ?: 0
        if (remaining <= 0) {
            return Result.failure(NoRollsRemainingException())
        }
        
        // Decrement roll count
        return subscriptionRepository.decrementDailyRolls()
            .map { Unit }
    }
}

class NoRollsRemainingException : Exception("No rolls remaining")
```

**Handling Roll Limits:**

```kotlin
viewModelScope.launch {
    rollDiceUseCase()
        .onSuccess {
            // Generate icebreaker
            generateIcebreaker()
        }
        .onFailure { error ->
            if (error is NoRollsRemainingException) {
                // Show paywall
                _uiState.update { it.copy(showPaywallNavigation = true) }
            } else {
                // Show error
                _uiState.update { it.copy(error = error.message) }
            }
        }
}
```

## SubscriptionCard Component

Display subscription status with a premium card:

```kotlin
@Composable
fun SubscriptionCard(
    modifier: Modifier = Modifier,
    status: SubscriptionStatus?,
    onUpgradeClick: () -> Unit
) {
    val tier = status?.tier ?: SubscriptionTier.FREE
    val rollsRemaining = status?.dailyRollsRemaining
    val isLifetime = status?.isLifetime ?: false
    val isPro = tier == SubscriptionTier.PREMIUM
    
    Box(
        modifier = modifier
            .fillMaxWidth()
            .clip(RoundedCornerShape(24.dp))
            .background(
                brush = Brush.linearGradient(
                    colors = if (isPro) {
                        listOf(NeonPink.copy(alpha = 0.8f), Color(0xFF8E24AA))
                    } else {
                        listOf(
                            Color.White.copy(alpha = 0.05f), 
                            Color.White.copy(alpha = 0.08f)
                        )
                    }
                )
            )
            .padding(24.dp)
    ) {
        Column {
            Row(
                verticalAlignment = Alignment.CenterVertically,
                horizontalArrangement = Arrangement.SpaceBetween,
                modifier = Modifier.fillMaxWidth()
            ) {
                Column {
                    Text(
                        text = "Current Plan",
                        style = MaterialTheme.typography.labelLarge,
                        color = TextWhite.copy(alpha = 0.6f)
                    )
                    
                    val tierText = when {
                        isLifetime -> "Lifetime Premium"
                        isPro -> "Premium"
                        else -> "Free"
                    }
                    
                    Text(
                        text = tierText,
                        style = MaterialTheme.typography.headlineSmall,
                        color = if (isPro) TextWhite else NeonCyan,
                        fontWeight = FontWeight.ExtraBold
                    )
                }
                
                Icon(
                    imageVector = Icons.Default.Star,
                    contentDescription = null,
                    tint = if (isPro) Color(0xFFFFD700) else TextWhite.copy(alpha = 0.2f),
                    modifier = Modifier.size(40.dp)
                )
            }
            
            Spacer(modifier = Modifier.height(16.dp))
            
            Text(
                text = if (isPro) {
                    "Unlimited rolls • All categories • Ad-free"
                } else {
                    "Limited daily rolls • Basic categories"
                },
                style = MaterialTheme.typography.bodyMedium,
                color = TextWhite.copy(alpha = 0.8f)
            )
            
            if (!isPro && rollsRemaining != null) {
                Spacer(modifier = Modifier.height(12.dp))
                Text(
                    text = "$rollsRemaining rolls left today",
                    style = MaterialTheme.typography.bodySmall,
                    color = NeonPink,
                    fontWeight = FontWeight.Bold
                )
            }
            
            Spacer(modifier = Modifier.height(24.dp))
            
            Button(
                onClick = onUpgradeClick,
                modifier = Modifier.fillMaxWidth(),
                shape = RoundedCornerShape(16.dp),
                colors = ButtonDefaults.buttonColors(
                    containerColor = if (isPro) 
                        Color.White.copy(alpha = 0.2f) 
                    else 
                        NeonCyan
                )
            ) {
                Text(
                    text = if (isPro) "Manage Subscription" else "Upgrade to Premium",
                    color = if (isPro) TextWhite else Color.Black,
                    fontWeight = FontWeight.Bold
                )
            }
        }
    }
}
```

## Best Practices

<Steps>
  <Step title="Require Authentication">
    Always require authentication before showing the paywall:

    ```kotlin
    val isAnonymous = authUiState.user?.isAnonymous ?: true
    if (isAnonymous) {
        // Show login prompt first
        showLoginDialog = true
    } else {
        // Show paywall
        showPaywall = true
    }
    ```
  </Step>

  <Step title="Observe Status Reactively">
    Use StateFlow to react to subscription changes:

    ```kotlin
    viewModelScope.launch {
        subscriptionRepository.getSubscriptionStatus().collect { status ->
            if (status.tier == SubscriptionTier.PREMIUM) {
                // Show confetti animation
                showCelebration()
            }
        }
    }
    ```
  </Step>

  <Step title="Handle Purchase Callbacks">
    Always refresh status after purchases:

    ```kotlin
    listener = object : PaywallListener {
        override fun onPurchaseCompleted(
            customerInfo: CustomerInfo,
            storeTransaction: StoreTransaction
        ) {
            viewModel.refreshStatus()
            // Show success feedback
        }
    }
    ```
  </Step>
</Steps>

<Warning>
  Never hardcode entitlement checks. Always use the `CheckEntitlementUseCase` to ensure server-side validation.
</Warning>
